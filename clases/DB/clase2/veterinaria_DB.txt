CREATE DATABASE veterinaria

go

USE veterinaria

CREATE TABLE duenio (
id_duenio int identity (1,1) PRIMARY KEY,
nombre NVARCHAR(80),
email NVARCHAR(180)
);

go

CREATE TABLE mascotas (
id_mascotas int identity(1,1)PRIMARY KEY,
nombre NCHAR(80) NOT NULL,
especie NCHAR(50) NOT NULL,
nacimiento DATE,
id_duenio int NOT NULL,
CONSTRAINT fk_duenio FOREIGN KEY (id_duenio) REFERENCES duenio (id_duenio)
);

go
